// ============================================

(function() {
    'use strict';

    // Variables globales del módulo

        assignee: '',
        priority: '',

        dateFrom: '',

    // ============================================

    function applyBoardStyles() {

        if (board) {

            board.style.minHeight = 'calc(100vh - 80px)';

    // ============================================
    // 📛 CARGAR NOMBRE DEL PROYECTO
    // ============================================
    function loadProjectName() {
        const nombreSpan = document.getElementById('projectNameBoard');
        if (nombreSpan && typeof window.projects !== 'undefined' && window.currentProjectIndex !== undefined) {
            const proyecto = window.projects[window.currentProjectIndex];
            if (proyecto && proyecto.name) {

                nombreSpan.title = proyecto.name;
            }

    // ============================================
    // 👥 POBLAR FILTRO DE ASIGNADOS
    // ============================================
    function populateAssigneeFilter() {
        const select = document.getElementById('filterAssigneeBoard');
        if (!select || typeof window.projects === 'undefined') return;
        
        const project = window.projects[window.currentProjectIndex];
        if (!project || !project.tasks) return;
        
        // Obtener asignados únicos
        const assignees = [...new Set(
            project.tasks
                .map(t => t.assignee)
                .filter(a => a && a.trim() !== '')
        )].sort();
        
        // Mantener opción "Todos"
        const currentValue = select.value;
        select.innerHTML = '<option value="">Todos</option>';
        
        assignees.forEach(assignee => {
            const option = document.createElement('option');
            option.value = assignee;
            option.textContent = assignee;
            select.appendChild(option);
        });
        
        // Restaurar valor seleccionado
        if (currentValue && assignees.includes(currentValue)) {
            select.value = currentValue;
        }
    }
    
    // ============================================
    // 🎴 RENDERIZAR TAREAS EN EL BOARD
    // ============================================
    function renderBoardTasks() {
        console.log('🎴 Renderizando tareas en Board...');
        
        const project = window.projects?.[window.currentProjectIndex];
        if (!project || !project.tasks) {
            showEmptyBoardMessage();
            return;
        }
        
        // Limpiar contenedores
        ['pending', 'inProgress', 'completed', 'overdue'].forEach(status => {
            const container = document.querySelector(`#${status}List .tasks-container`);
            if (container) container.innerHTML = '';
        });
        
        // Filtrar tareas según filtros activos
        const filteredTasks = filterTasks(project.tasks);
        
        if (filteredTasks.length === 0) {
            showEmptyBoardMessage();
            updateColumnCounts();
            return;
        }
        
        hideEmptyBoardMessage();
        
        // Renderizar cada tarea en su columna correspondiente
        filteredTasks.forEach(task => {
            const taskElement = createTaskCard(task);
            const column = document.querySelector(`#${task.status}List .tasks-container`);
            if (column) {
                column.appendChild(taskElement);
            }
        });
        
        // Actualizar contadores de columnas
        updateColumnCounts();
        
        // Reinicializar drag & drop
        setupNativeDragDrop();
        
        console.log(`✅ ${filteredTasks.length} tareas renderizadas`);
    }
    
    // ============================================
    // 🔍 FILTRAR TAREAS
    // ============================================
    function filterTasks(tasks) {
        return tasks.filter(task => {
            // Filtro por asignado
            if (currentFilters.assignee && task.assignee !== currentFilters.assignee) {
                return false;
            }
            
            // Filtro por prioridad
            if (currentFilters.priority && task.priority !== currentFilters.priority) {
                return false;
            }
            
            // Filtro por estado
            if (currentFilters.status && task.status !== currentFilters.status) {
                return false;
            }
            
            // Filtro por fecha desde
            if (currentFilters.dateFrom && task.startDate) {
                const taskDate = new Date(task.startDate);
                const filterDate = new Date(currentFilters.dateFrom);
                if (taskDate < filterDate) return false;
            }
            
            // Filtro por fecha hasta
            if (currentFilters.dateTo && task.deadline) {
                const taskDate = new Date(task.deadline);
                const filterDate = new Date(currentFilters.dateTo);
                if (taskDate > filterDate) return false;
            }
            
            return true;
        });
    }
    
    // ============================================
    // 🃏 CREAR TARJETA DE TAREA
    // ============================================
    function createTaskCard(task) {
        const card = document.createElement('div');
        card.className = 'task-card';
        card.setAttribute('draggable', 'true');
        card.setAttribute('data-task-id', task.id);
        card.setAttribute('data-status', task.status);
        
        // Contenido de la tarjeta
        card.innerHTML = `
            <div class="task-controls">
                <button class="task-menu" title="Más opciones" onclick="toggleTaskMenu(event, '${task.id}')">⋮</button>
            </div>
            
            <div class="task-title">${escapeHtml(task.name || 'Sin título')}</div>
            
            ${task.priority ? `<span class="priority-badge ${task.priority}">${getPriorityLabel(task.priority)}</span>` : ''}
            
            ${task.assignee ? `
                <div class="task-assignee">
                    <i class="fas fa-user"></i>
                    <span>${escapeHtml(task.assignee)}</span>
                </div>
            ` : ''}
            
            ${task.deadline ? `
                <div class="task-deadline ${isOverdue(task) ? 'overdue' : ''}">
                    <i class="fas fa-calendar"></i>
                    <span>${formatDate(task.deadline)}</span>
                </div>
            ` : ''}
            
            ${task.subtasks && task.subtasks.length > 0 ? createSubtasksProgress(task.subtasks) : ''}
            
            <!-- Menú contextual -->
            <div id="menu-${task.id}" class="task-context-menu">
                <div class="task-context-menu-item" onclick="editTask('${task.id}')">
                    <i class="fas fa-edit"></i> Editar
                </div>
                <div class="task-context-menu-item" onclick="viewTaskDetails('${task.id}')">
                    <i class="fas fa-eye"></i> Ver detalles
                </div>
                <div class="task-context-menu-item delete" onclick="deleteTask('${task.id}')">
                    <i class="fas fa-trash"></i> Eliminar
                </div>
            </div>
        `;
        
        // Eventos de la tarjeta
        card.addEventListener('dragstart', handleDragStart);
        card.addEventListener('dragend', handleDragEnd);
        card.addEventListener('click', (e) => {
            if (!e.target.closest('.task-menu') && !e.target.closest('.task-context-menu')) {
                viewTaskDetails(task.id);
            }
        });
        
        return card;
    }
    
    // ============================================
    // 📊 PROGRESO DE SUBTAREAS
    // ============================================
    function createSubtasksProgress(subtasks) {
        const total = subtasks.length;
        const completed = subtasks.filter(s => s.completed).length;
        const percentage = total > 0 ? Math.round((completed / total) * 100) : 0;
        
        return `
            <div class="subtasks-progress-container">
                <div class="progress-bar">
                    <div class="progress-fill" style="width: ${percentage}%"></div>
                </div>
                <div class="progress-text">${completed}/${total} subtareas • ${percentage}%</div>
            </div>
        `;
    }
    
    // ============================================
    // 🔢 ACTUALIZAR CONTADORES DE COLUMNAS

        ['pending', 'inProgress', 'completed', 'overdue'].forEach(status => {
            const container = document.querySelector(`#${status}List .tasks-container`);
            const countSpan = document.getElementById(`${status}Count`);

                const count = container.querySelectorAll('.task-card').length;
                countSpan.textContent = count;
                countSpan.setAttribute('aria-label', `${count} tareas ${getStatusLabel(status)}`);

    function setupEventListeners() {

    // ============================================
    // 🖱️ DRAG & DROP NATIVO
    // ============================================
    function setupNativeDragDrop() {
        // Configurar tareas como arrastrables
        document.querySelectorAll('.task-card').forEach(card => {
            card.setAttribute('draggable', 'true');
            card.ondragstart = handleDragStart;
            card.ondragend = handleDragEnd;
        });
        
        // Configurar zonas de drop
        document.querySelectorAll('.tasks-container').forEach(container => {
            container.ondragover = handleDragOver;
            container.ondragleave = handleDragLeave;
            container.ondrop = handleDrop;
        });
    }
    
    function handleDragStart(e) {
        e.dataTransfer.setData('text/plain', e.target.dataset.taskId);
        e.target.classList.add('dragging');
        e.dataTransfer.effectAllowed = 'move';
    }
    
    function handleDragEnd(e) {
        e.target.classList.remove('dragging');
    }
    
    function handleDragOver(e) {
        e.preventDefault();
        e.dataTransfer.dropEffect = 'move';
        e.currentTarget.classList.add('drag-over');
    }
    
    function handleDragLeave(e) {
        e.currentTarget.classList.remove('drag-over');
    }
    
    async function handleDrop(e, newStatus) {
        e.preventDefault();
        e.currentTarget.classList.remove('drag-over');
        
        const taskId = e.dataTransfer.getData('text/plain');
        if (!taskId) return;
        
        const taskCard = document.querySelector(`[data-task-id="${taskId}"]`);
        if (!taskCard) return;
        
        console.log(`🔄 Moviendo tarea ${taskId} a ${newStatus}`);
        
        try {
            // Actualizar estado en datos
            const project = window.projects?.[window.currentProjectIndex];
            if (project && project.tasks) {
                const task = project.tasks.find(t => t.id == taskId);
                if (task) {
                    const oldStatus = task.status;
                    task.status = newStatus;
                    
                    // Actualizar progreso según estado
                    if (newStatus === 'completed') {
                        task.progress = 100;
                    } else if (newStatus === 'inProgress') {
                        task.progress = task.progress || 50;
                    } else {
                        task.progress = 0;
                    }
                    
                    // Guardar cambios
                    if (typeof window.safeSave === 'function') {
                        await window.safeSave();
                    }
                    
                    // Actualizar UI
                    taskCard.setAttribute('data-status', newStatus);
                    e.currentTarget.appendChild(taskCard);
                    updateColumnCounts();
                    
                    // Feedback
                    showNotification(`Tarea movida a "${getStatusLabel(newStatus)}"`, 'success');
                    
                    // Actualizar burndown si está visible
                    if (typeof window.updateBurndownOnStatusChange === 'function') {
                        window.updateBurndownOnStatusChange();
                    }
                }
            }
        } catch (error) {
            console.error('❌ Error moviendo tarea:', error);
            showNotification('Error al mover tarea', 'error');
        }
    }
    
    // ============================================
    // 🎛️ MENÚ CONTEXTUAL DE TAREA
    // ============================================
    window.toggleTaskMenu = function(event, taskId) {
        event.stopPropagation();
        
        // Cerrar todos los menús primero
        document.querySelectorAll('.task-context-menu').forEach(menu => {
            if (menu.id !== `menu-${taskId}`) {
                menu.classList.remove('show');
            }
        });
        
        // Toggle del menú actual
        const menu = document.getElementById(`menu-${taskId}`);
        if (menu) {
            const isVisible = menu.classList.contains('show');
            document.querySelectorAll('.task-context-menu').forEach(m => m.classList.remove('show'));
            if (!isVisible) {
                menu.classList.add('show');
                // Posicionar menú
                const rect = event.target.getBoundingClientRect();
                menu.style.top = `${rect.bottom + 5}px`;
                menu.style.left = `${rect.right - 160}px`;
            }
        }
    };
    
    // ============================================
    // ✏️ EDITAR TAREA
    // ============================================
    window.editTask = async function(taskId) {
        console.log('✏️ Editando tarea:', taskId);
        
        const project = window.projects?.[window.currentProjectIndex];
        const task = project?.tasks?.find(t => t.id == taskId);
        
        if (!task) {
            showNotification('Tarea no encontrada', 'error');
            return;
        }
        
        // Cerrar menú contextual
        document.querySelectorAll('.task-context-menu').forEach(m => m.classList.remove('show'));
        
        // Abrir modal de edición si existe
        if (typeof window.showEditTaskModal === 'function') {
            window.showEditTaskModal(task);
        } else {
            // Fallback: mostrar detalles
            viewTaskDetails(taskId);
        }
    };
    
    // ============================================
    // 👁️ VER DETALLES DE TAREA
    // ============================================
    window.viewTaskDetails = function(taskId) {
        console.log('👁️ Viendo detalles de tarea:', taskId);
        
        if (typeof window.showTaskDetails === 'function') {
            window.showTaskDetails(taskId);
        } else if (typeof window.openTaskModal === 'function') {
            window.openTaskModal(taskId);
        } else {
            // Fallback simple
            const project = window.projects?.[window.currentProjectIndex];
            const task = project?.tasks?.find(t => t.id == taskId);
            if (task) {
                alert(`📋 ${task.name}\n\nEstado: ${getStatusLabel(task.status)}\nPrioridad: ${getPriorityLabel(task.priority)}\nAsignado: ${task.assignee || 'Sin asignar'}\nFecha límite: ${task.deadline ? formatDate(task.deadline) : 'Sin fecha'}`);
            }
        }
    };
    
    // ============================================
    // 🗑️ ELIMINAR TAREA
    // ============================================
    window.deleteTask = async function(taskId) {
        if (!confirm('¿Estás seguro de eliminar esta tarea?\n\nEsta acción no se puede deshacer.')) {
            return;
        }
        
        console.log('🗑️ Eliminando tarea:', taskId);
        
        try {
            const project = window.projects?.[window.currentProjectIndex];
            if (project && project.tasks) {
                const index = project.tasks.findIndex(t => t.id == taskId);
                if (index !== -1) {
                    const taskName = project.tasks[index].name;
                    project.tasks.splice(index, 1);
                    
                    // Guardar cambios
                    if (typeof window.safeSave === 'function') {
                        await window.safeSave();
                    }
                    
                    // Actualizar UI
                    renderBoardTasks();
                    showNotification(`Tarea "${taskName}" eliminada`, 'success');
                }
            }
        } catch (error) {
            console.error('❌ Error eliminando tarea:', error);
            showNotification('Error al eliminar tarea', 'error');
        }
    };

    // ============================================
    window.exportBoardToPDF = async function() {
        console.log('📄 Exportando Board a PDF...');
        
        showNotification('Generando PDF...', 'info');
        
        try {
            // Verificar librería jsPDF
            if (typeof window.jspdf === 'undefined' && typeof window.jsPDF === 'undefined') {
                throw new Error('jsPDF no está cargado');
            }
            
            const { jsPDF } = window.jspdf || window;
            const doc = new jsPDF('landscape', 'mm', 'a4');
            
            const project = window.projects?.[window.currentProjectIndex];
            if (!project) throw new Error('No hay proyecto seleccionado');
            
            // Título
            doc.setFontSize(18);
            doc.setTextColor(44, 62, 80);
            doc.text(`Board Kanban - ${project.name}`, 14, 20);
            
            // Fecha
            doc.setFontSize(10);
            doc.setTextColor(100, 116, 139);
            doc.text(`Generado: ${new Date().toLocaleString('es-ES')}`, 14, 28);
            
            // Columnas
            const columns = [
                { id: 'pending', label: 'PENDIENTES', color: [241, 196, 15] },
                { id: 'inProgress', label: 'EN PROGRESO', color: [0, 128, 144] },
                { id: 'completed', label: 'COMPLETADOS', color: [46, 204, 113] },
                { id: 'overdue', label: 'REZAGADOS', color: [231, 76, 60] }
            ];
            
            let yPos = 40;
            
            columns.forEach((col, colIndex) => {
                const container = document.querySelector(`#${col.id}List .tasks-container`);
                const tasks = container?.querySelectorAll('.task-card') || [];
                
                // Título de columna
                doc.setFillColor(...col.color);
                doc.rect(14 + (colIndex * 70), yPos - 8, 65, 10, 'F');
                doc.setFontSize(11);
                doc.setTextColor(255, 255, 255);
                doc.text(`${col.label} (${tasks.length})`, 17 + (colIndex * 70), yPos - 1);
                
                // Tareas
                doc.setFontSize(9);
                doc.setTextColor(44, 62, 80);
                
                Array.from(tasks).slice(0, 8).forEach((taskEl, taskIndex) => {
                    const title = taskEl.querySelector('.task-title')?.textContent || 'Sin título';
                    const truncated = title.length > 35 ? title.substring(0, 32) + '...' : title;
                    
                    doc.text(`• ${truncated}`, 17 + (colIndex * 70), yPos + 5 + (taskIndex * 6));
                });
                
                if (tasks.length > 8) {
                    doc.setTextColor(100, 116, 139);
                    doc.text(`+ ${tasks.length - 8} más...`, 17 + (colIndex * 70), yPos + 5 + (8 * 6));
                }
            });
            
            // Footer
            doc.setFontSize(8);
            doc.setTextColor(150, 150, 150);
            doc.text('Sistema de Gestión de Proyectos - Zacky', 14, 190);
            
            // Descargar
            const fileName = `Board-${project.name.replace(/[^a-z0-9]/gi, '_')}-${new Date().toISOString().slice(0,10)}.pdf`;
            doc.save(fileName);
            
            showNotification('✅ PDF exportado exitosamente', 'success');
            
        } catch (error) {
            console.error('❌ Error exportando PDF:', error);
            showNotification('Error al exportar PDF: ' + error.message, 'error');
        }
    };

                x: 0.5, y: 1.5, w: 4,
                colW: [3, 1],
                fill: 'F8F9FA',
                color: '2C3E50',
                fontSize: 12
            });
            
            // Slide 3: Tareas por columna
            slide = pptx.addSlide();
            slide.background = { fill: { color: 'FFFFFF' } };
            
            slide.addText('TAREAS POR ESTADO', {
                x: 0.5, y: 0.5, w: 9, h: 0.8,
                fontSize: 20, bold: true, color: '2C3E50'
            });
            
            const columns = ['pending', 'inProgress', 'completed', 'overdue'];
            let yPos = 1.8;
            
            columns.forEach((status, index) => {
                const container = document.querySelector(`#${status}List .tasks-container`);
                const tasks = Array.from(container?.querySelectorAll('.task-card') || []);
                
                slide.addText(`${getStatusLabel(status).toUpperCase()} (${tasks.length})`, {
                    x: 0.5 + (index * 2.4), y: yPos - 0.3, w: 2.2, h: 0.4,
                    fontSize: 11, bold: true, color: getColumnColor(status)
                });
                
                tasks.slice(0, 6).forEach((taskEl, taskIndex) => {
                    const title = taskEl.querySelector('.task-title')?.textContent || 'Sin título';
                    const truncated = title.length > 25 ? title.substring(0, 22) + '...' : title;
                    
                    slide.addText(`• ${truncated}`, {
                        x: 0.5 + (index * 2.4),
                        y: yPos + (taskIndex * 0.4),
                        w: 2.2,
                        fontSize: 9,
                        color: '2C3E50'
                    });
                });
            });
            
            // Descargar
            const fileName = `Board-${project.name.replace(/[^a-z0-9]/gi, '_')}-${new Date().toISOString().slice(0,10)}.pptx`;
            await pptx.writeFile({ fileName: fileName });
            
            showNotification('✅ PowerPoint exportado exitosamente', 'success');
            
        } catch (error) {
            console.error('❌ Error exportando PowerPoint:', error);
            showNotification('Error al exportar: ' + error.message, 'error');
        }
    };
    
    // ============================================
    // 📊 ESTADÍSTICAS DEL BOARD
    // ============================================
    function getBoardStats() {
        const project = window.projects?.[window.currentProjectIndex];
        if (!project || !project.tasks) {
            return { total: 0, pending: 0, inProgress: 0, completed: 0, overdue: 0 };
        }
        
        const tasks = project.tasks;
        return {
            total: tasks.length,
            pending: tasks.filter(t => t.status === 'pending').length,
            inProgress: tasks.filter(t => t.status === 'inProgress').length,
            completed: tasks.filter(t => t.status === 'completed').length,
            overdue: tasks.filter(t => t.status === 'overdue').length
        };
    }
    
    // ============================================
    // 🎨 UTILIDADES DE FORMATO
    // ============================================
    function getStatusLabel(status) {
        const labels = {
            pending: 'Pendiente',
            inProgress: 'En Progreso',
            completed: 'Completado',
            overdue: 'Rezagado'
        };
        return labels[status] || status;
    }
    
    function getPriorityLabel(priority) {
        const labels = {
            baja: 'Baja',
            media: 'Media',
            alta: 'Alta'
        };
        return labels[priority] || priority;
    }
    
    function getColumnColor(status) {
        const colors = {
            pending: 'F1C40F',
            inProgress: '008090',
            completed: '2ECC71',
            overdue: 'E74C3C'
        };
        return colors[status] || '64748B';
    }
    
    function formatDate(dateString) {
        if (!dateString) return '';
        try {
            const date = new Date(dateString);
            return date.toLocaleDateString('es-ES', { day: '2-digit', month: 'short' });
        } catch {
            return dateString;
        }
    }
    
    function isOverdue(task) {
        if (!task.deadline || task.status === 'completed') return false;
        return new Date(task.deadline) < new Date();
    }
    
    function escapeHtml(text) {
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }
    
    // ============================================
    // 🔔 NOTIFICACIONES
    // ============================================
    function showNotification(message, type = 'info') {
        if (typeof window.showNotification === 'function') {
            window.showNotification(message, type);
        } else {
            // Fallback simple
            console.log(`[${type.toUpperCase()}] ${message}`);
            // Podrías implementar un toast simple aquí si lo necesitas
        }
    }
    
    // ============================================
    // 🔄 REFRESCAR BOARD
    // ============================================
    window.refreshBoard = function() {
        console.log('🔄 Refrescando Board...');
        populateAssigneeFilter();
        renderBoardTasks();
    };
    
    // ============================================
    // 🚀 INICIALIZAR CUANDO EL DOM ESTÉ LISTO
    // ============================================
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', initBoard);
    } else {
        initBoard();
    }
    
    // Escuchar cambios en proyectos para auto-refrescar
    if (window.addEventListener) {
        window.addEventListener('projectUpdated', () => {
            console.log('🔄 Proyecto actualizado, refrescando Board...');
            setTimeout(renderBoardTasks, 100);
        });
    }
    
})();

<script>
// ============================================
// 🔥 SOLUCIÓN DEFINITIVA - FILTROS DEL BOARD

(function() {
    console.log('🔥 ACTIVANDO FILTROS DEL BOARD...');
    
    // Función para asegurar que los proyectos estén cargados
    function ensureProjects() {
        if (!window.projects || window.projects.length === 0) {
            const projectsJSON = localStorage.getItem('projects');
            if (projectsJSON) {
                try {
                    window.projects = JSON.parse(projectsJSON);
                    console.log('✅ Proyectos cargados desde localStorage');
                } catch(e) {}
            }
        }
        
        if (!window.currentProjectIndex && window.currentProjectIndex !== 0) {
            window.currentProjectIndex = 0;
        }
        
        return window.projects && window.projects[window.currentProjectIndex];
    }
    
    // Función para poblar el select de asignados
    function populateAssigneeFilter() {
        const select = document.getElementById('filterAssigneeBoard');
        if (!select) return;
        
        const project = ensureProjects();
        if (!project || !project.tasks) return;
        
        // Obtener asignados únicos
        const assignees = [...new Set(
            project.tasks
                .map(t => t.assignee)
                .filter(a => a && a.trim() !== '')
        )].sort();
        
        console.log('👥 Asignados encontrados:', assignees);
        
        // Poblar select
        select.innerHTML = '<option value="">Todos</option>';
        assignees.forEach(assignee => {
            const option = document.createElement('option');
            option.value = assignee;
            option.textContent = assignee;
            select.appendChild(option);
        });
    }
    
    // Función para filtrar usando EL MISMO SISTEMA QUE EL BOARD
    function filterBoardTasks() {
        console.log('🔍 FILTRANDO TAREAS...');
        
        const assignee = document.getElementById('filterAssigneeBoard')?.value || '';
        const priority = document.getElementById('filterPriorityBoard')?.value || '';
        const status = document.getElementById('filterStatusBoard')?.value || '';
        const dateFrom = document.getElementById('filterDateFromBoard')?.value || '';
        const dateTo = document.getElementById('filterDateToBoard')?.value || '';
        
        console.log('Filtros:', { assignee, priority, status, dateFrom, dateTo });
        
        const project = ensureProjects();
        if (!project || !project.tasks) {
            console.error('❌ No hay proyecto');
            return;
        }
        
        // Guardar tareas originales
        const originalTasks = [...project.tasks];
        
        // Aplicar filtros
        const filteredTasks = originalTasks.filter(task => {
            if (assignee && task.assignee !== assignee) return false;
            if (priority && task.priority !== priority) return false;
            if (status && task.status !== status) return false;
            
            if (dateFrom && task.deadline) {
                try {
                    if (new Date(task.deadline) < new Date(dateFrom)) return false;
                } catch(e) {}
            }
            
            if (dateTo && task.deadline) {
                try {
                    if (new Date(task.deadline) > new Date(dateTo)) return false;
                } catch(e) {}
            }
            
            return true;
        });
        
        console.log(`📊 Mostrando ${filteredTasks.length} de ${originalTasks.length} tareas`);
        
        // REEMPLAZAR TEMPORALMENTE Y RENDERIZAR
        project.tasks = filteredTasks;
        
        // USAR LA FUNCIÓN EXISTENTE DEL SISTEMA
        if (typeof window.renderKanbanTasks === 'function') {
            window.renderKanbanTasks();
            console.log('✅ renderKanbanTasks ejecutado');
        } else {
            console.error('❌ No se encontró renderKanbanTasks');
        }
        
        // Restaurar después de 3 segundos
        setTimeout(() => {
            project.tasks = originalTasks;
            if (typeof window.renderKanbanTasks === 'function') {
                window.renderKanbanTasks();
            }
            console.log('🔄 Tareas restauradas');
        }, 3000);
    }
    
    // Función para limpiar filtros
    function clearBoardFilters() {
        console.log('🧹 Limpiando filtros');
        
        document.getElementById('filterAssigneeBoard').value = '';
        document.getElementById('filterPriorityBoard').value = '';
        document.getElementById('filterStatusBoard').value = '';
        document.getElementById('filterDateFromBoard').value = '';
        document.getElementById('filterDateToBoard').value = '';
        
        // Renderizar todas las tareas
        const project = ensureProjects();
        if (project && typeof window.renderKanbanTasks === 'function') {
            window.renderKanbanTasks();
        }
    }
    
    // Configurar eventos
    function setupEvents() {
        console.log('🔌 Configurando eventos...');
        
        const applyBtn = document.getElementById('applyFiltersBoard');
        const clearBtn = document.getElementById('clearFiltersBoard');
        
        // Eventos de cambio automático
        document.getElementById('filterAssigneeBoard')?.addEventListener('change', filterBoardTasks);
        document.getElementById('filterPriorityBoard')?.addEventListener('change', filterBoardTasks);
        document.getElementById('filterStatusBoard')?.addEventListener('change', filterBoardTasks);
        document.getElementById('filterDateFromBoard')?.addEventListener('change', filterBoardTasks);
        document.getElementById('filterDateToBoard')?.addEventListener('change', filterBoardTasks);
        
        // Botones
        if (applyBtn) applyBtn.addEventListener('click', filterBoardTasks);
        if (clearBtn) clearBtn.addEventListener('click', clearBoardFilters);
        
        console.log('✅ Eventos configurados');
    }
    
    // Inicializar
    function init() {
        console.log('🚀 Inicializando filtros...');
        
        // Cargar proyectos si es necesario
        ensureProjects();
        
        // Poblar filtros
        populateAssigneeFilter();
        
        // Configurar eventos
        setupEvents();
        
        // Observar cambios en el proyecto
        document.addEventListener('projectChanged', () => {
            setTimeout(populateAssigneeFilter, 200);
        });
        
        // Observar cuando se muestra el board
        const boardView = document.getElementById('boardView');
        if (boardView) {
            const observer = new MutationObserver(() => {
                if (boardView.style.display !== 'none') {
                    setTimeout(populateAssigneeFilter, 300);
                }
            });
            observer.observe(boardView, { attributes: true });
        }
    }
    
    // Ejecutar cuando el DOM esté listo
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
    } else {
        init();
    }
    
    // Exponer funciones para pruebas
    window.testBoardFilters = function() {

            console.log('📌 Estados:', [...new Set(project.tasks.map(t => t.status).filter(Boolean))]);
        }
    };
    
    console.log('✅ FILTROS LISTOS - Usa testBoardFilters() en consola');
})();

showNotification('✅ Transcripción eliminada');
if (typeof cargarTranscripciones === 'function') {
cargarTranscripciones();

}
} else {

}
} catch (error) {
console.error('❌ Error:', error);
alert('❌ Error de conexión');

window.eliminarTodasTranscripciones = async function() {
if (!confirm('¿Eliminar TODAS las transcripciones?
Esta acción no se puede deshacer.')) return;
console.log('🗑️ Eliminando todas las transcripciones');
const clienteId = localStorage.getItem('clienteId');
const response = await fetch(`https://mi-sistema-proyectos-backend-4.onrender.com/api/transcripciones?clienteId=${clienteId}`);
const data = await response.json();
if (!data.transcripciones || data.transcripciones.length === 0) {
alert('📭 No hay transcripciones para eliminar');
return;
}
let eliminadas = 0;
let errores = 0;
for (const t of data.transcripciones) {
try {

method: 'DELETE'
});
const delData = await delResponse.json();
if (delData.success) eliminadas++;
else errores++;
} catch (e) {
errores++;
}
}
alert(`✅ ${eliminadas} transcripciones eliminadas${errores > 0 ? `, ${errores} errores` : ''}`);
if (typeof cargarTranscripciones === 'function') {
cargarTranscripciones();
} else {
location.reload();
}
};
console.log('✅ Funciones de eliminación cargadas desde HTML');

// Función auxiliar para formatear fechas
function formatDate(dateString) {
if (!dateString) return '--/--/--';
const date = new Date(dateString);
if (isNaN(date.getTime())) return '--/--/--';
const day = String(date.getDate()).padStart(2, '0');
const month = String(date.getMonth() + 1).padStart(2, '0');
const year = date.getFullYear();
return `${day}/${month}/${year}`;
}

const year = earliestDate.getFullYear();
document.getElementById('projectStartDate').textContent = `${day}/${month}/${year}`;
} else {
document.getElementById('projectStartDate').textContent = '--/--/--';
}
}
const project = projects[currentProjectIndex];
if (!project || !project.tasks || project.tasks.length === 0) {
document.getElementById('projectStartDate').textContent = '--/--/--';
document.getElementById('projectEndDate').textContent = '--/--/--';
return;
}
const tasks = project.tasks;
// Extraer y filtrar fechas de inicio (startDate) y fin (deadline)
const startDates = tasks
.filter(t => t.startDate)
.map(t => new Date(t.startDate))
.filter(date => !isNaN(date.getTime()));
const endDates = tasks
.filter(t => t.deadline)
.map(t => new Date(t.deadline))
.filter(date => !isNaN(date.getTime()));
// Si no hay fechas válidas
if (startDates.length === 0) {
document.getElementById('projectStartDate').textContent = '--/--/--';
} else {
const earliestStart = new Date(Math.min(...startDates));
document.getElementById('projectStartDate').textContent = formatDate(earliestStart);
}
if (endDates.length === 0) {
document.getElementById('projectEndDate').textContent = '--/--/--';
} else {
const latestEnd = new Date(Math.max(...endDates));
document.getElementById('projectEndDate').textContent = formatDate(latestEnd);
}
}

function getFilteredCalendarTasks() {
const assignee = document.getElementById('filterAssigneeCalendar')?.value.trim();
const priority = document.getElementById('filterPriorityCalendar')?.value;
const status = document.getElementById('filterStatusCalendar')?.value;
const tasks = projects[currentProjectIndex]?.tasks || [];
console.log('📅 Tareas para calendario:', tasks.length);
const filteredTasks = tasks
.filter(task => {
const taskAssignee = task.assignee?.trim() || '';
const taskPriority = task.priority || '';
const taskStatus = task.status || '';
const matchAssignee = !assignee || taskAssignee === assignee;
const matchPriority = !priority || taskPriority === priority;
const matchStatus = !status || taskStatus === status;
return matchAssignee && matchPriority && matchStatus;
})
.map(task => {
// Determinar color según estado (gradientes)
const statusColors = {
'pending': 'linear-gradient(135deg, #f1c40f, #f39c12)',
'inProgress': 'linear-gradient(135deg, #008090, #1abc9c)',
'completed': 'linear-gradient(135deg, #2ecc71, #27ae60)',
'overdue': 'linear-gradient(135deg, #e74c3c, #c0392b)'
};
// Calcular días restantes para overdue
const today = new Date();
today.setHours(0, 0, 0, 0);
let daysRemaining = null;
let isOverdue = false;
if (task.deadline) {
const deadlineDate = new Date(task.deadline);
deadlineDate.setHours(0, 0, 0, 0);
const diffTime = deadlineDate - today;
const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
daysRemaining = diffDays;
isOverdue = diffDays < 0 && task.status !== 'completed';
}
// Usar fechas de inicio y fin
let startDate = task.startDate || task.deadline || new Date().toISOString().split('T')[0];
let endDate = task.deadline || task.startDate;
if (!endDate || endDate === startDate) {
endDate = null;
} else {
const end = new Date(endDate);
end.setDate(end.getDate() + 1);
endDate = end.toISOString().split('T')[0];
}
// Crear tooltip con HTML enriquecido
const tooltip = `
<div style="min-width: 250px;">
<div style="display:flex; align-items:center; gap:12px; margin-bottom:12px; padding-bottom:10px; border-bottom:1px solid rgba(255,255,255,0.1);">
<div style="width:40px; height:40px; background:linear-gradient(135deg,#8b5cf6,#6d28d9); border-radius:10px; display:flex; align-items:center; justify-content:center; font-size:18px;">📋</div>

<div style="color:#94a3b8; font-size:12px; margin-top:3px;">ID: ${task.id || 'N/A'}</div>
</div>
</div>
<div style="display:grid; grid-template-columns:repeat(2,1fr); gap:12px; margin-bottom:12px;">
<div style="background:rgba(255,255,255,0.05); padding:10px; border-radius:10px;">
<div style="color:#94a3b8; font-size:11px; margin-bottom:4px;">ASIGNADO A</div>
<div style="color:white; font-weight:600; font-size:14px;">${task.assignee || 'No asignado'}</div>
</div>
<div style="background:rgba(255,255,255,0.05); padding:10px; border-radius:10px;">
<div style="color:#94a3b8; font-size:11px; margin-bottom:4px;">PRIORIDAD</div>
<div style="color:white; font-weight:600; font-size:14px; text-transform:capitalize;">${task.priority || 'Media'}</div>
</div>
</div>
<div style="display:grid; grid-template-columns:repeat(2,1fr); gap:12px; margin-bottom:12px;">
<div style="background:rgba(255,255,255,0.05); padding:10px; border-radius:10px;">
<div style="color:#94a3b8; font-size:11px; margin-bottom:4px;">TIEMPO</div>

</div>
<div style="background:rgba(255,255,255,0.05); padding:10px; border-radius:10px;">
<div style="color:#94a3b8; font-size:11px; margin-bottom:4px;">PROGRESO</div>
<div style="color:white; font-weight:600; font-size:14px;">${task.progress || 0}%</div>
</div>
</div>
${daysRemaining !== null ? `
<div style="background:${isOverdue ? 'rgba(239,68,68,0.2)' : 'rgba(16,185,129,0.2)'}; padding:10px; border-radius:10px; text-align:center; border:1px solid ${isOverdue ? '#ef4444' : '#10b981'};">
<span style="color:${isOverdue ? '#ef4444' : '#10b981'}; font-weight:700;">
${isOverdue ? `⚠️ ATRASADA (${Math.abs(daysRemaining)} días)` : `✅ ${daysRemaining} días restantes`}
</span>
</div>
` : ''}
${task.description ? `
<div style="margin-top:12px; padding:10px; background:rgba(255,255,255,0.05); border-radius:10px; color:#94a3b8; font-size:12px; border-left:3px solid #8b5cf6;">
${task.description}
</div>
` : ''}
</div>
`;
return {
title: task.name,
start: startDate,

isOverdue
}
};
});
console.log('✅ Tareas procesadas para calendario:', filteredTasks.length);
return filteredTasks;
}

// 🔥 PARCHE DE EMERGENCIA - FILTROS DEL BOARD

(function() {
    console.log('🚑 APLICANDO PARCHE DE EMERGENCIA...');
    
    // Función para obtener el proyecto actual
    function getProject() {
        return window.projects?.[window.currentProjectIndex];
    }
    
    // Función para poblar el select de asignados
    function populateFilters() {
        const select = document.getElementById('filterAssigneeBoard');
        if (!select) return;
        
        const project = getProject();
        if (!project?.tasks) return;
        
        // Obtener asignados únicos
        const assignees = [];
        project.tasks.forEach(t => {
            if (t.assignee && t.assignee.trim() && !assignees.includes(t.assignee)) {
                assignees.push(t.assignee);
            }
        });
        
        console.log('📋 Asignados disponibles:', assignees);
        
        // Poblar select
        select.innerHTML = '<option value="">Todos</option>';
        assignees.sort().forEach(a => {
            const option = document.createElement('option');
            option.value = a;
            option.textContent = a;
            select.appendChild(option);
        });
    }
    
    // SOBRESCRIBIR la función que detecta cambios de filtro
    const originalFilterHandler = window.handleFilterChange || function() {};
    window.handleFilterChange = function(filterName, value) {
        console.log('🎯 FILTRO CAMBIADO - Procesando:', filterName, value);
        
        // Obtener valores actuales de todos los filtros
        const filters = {
            assignee: document.getElementById('filterAssigneeBoard')?.value || '',
            priority: document.getElementById('filterPriorityBoard')?.value || '',
            status: document.getElementById('filterStatusBoard')?.value || '',
            dateFrom: document.getElementById('filterDateFromBoard')?.value,
            dateTo: document.getElementById('filterDateToBoard')?.value
        };
        
        console.log('📊 Filtros actuales:', filters);
        
        const project = getProject();
        if (!project?.tasks) return;
        
        // Guardar tareas originales
        const originalTasks = [...project.tasks];
        
        // Aplicar filtros
        const filteredTasks = originalTasks.filter(task => {
            if (filters.assignee && task.assignee !== filters.assignee) return false;
            if (filters.priority && task.priority !== filters.priority) return false;
            if (filters.status && task.status !== filters.status) return false;
            
            if (filters.dateFrom && task.deadline) {
                try { if (new Date(task.deadline) < new Date(filters.dateFrom)) return false; } catch(e) {}
            }
            
            if (filters.dateTo && task.deadline) {
                try { if (new Date(task.deadline) > new Date(filters.dateTo)) return false; } catch(e) {}
            }
            
            return true;
        });
        
        console.log(`📈 Mostrando ${filteredTasks.length} de ${originalTasks.length} tareas`);
        
        // REEMPLAZAR TEMPORALMENTE Y RENDERIZAR
        const backup = project.tasks;
        project.tasks = filteredTasks;
        
        if (typeof window.renderKanbanTasks === 'function') {
            window.renderKanbanTasks();
            console.log('✅ Board actualizado');
        }
        
        // Restaurar (importante para no perder datos)
        project.tasks = backup;
    };

        console.log('🔌 Conectando filtros...');
        
        const elements = [
            'filterAssigneeBoard',
            'filterPriorityBoard', 
            'filterStatusBoard',
            'filterDateFromBoard',
            'filterDateToBoard'

                el.parentNode.replaceChild(newEl, el);
                
                // Agregar nuevo evento

                    window.handleFilterChange(this.id, this.value);
                });
                console.log(`✅ Evento conectado a ${id}`);
            }
        });

        if (applyBtn) {
            const newBtn = applyBtn.cloneNode(true);
            applyBtn.parentNode.replaceChild(newBtn, applyBtn);
            newBtn.addEventListener('click', function() {
                window.handleFilterChange('apply', 'click');
            });
            console.log('✅ Botón aplicar conectado');
        }

        if (clearBtn) {
            const newBtn = clearBtn.cloneNode(true);

                document.getElementById('filterPriorityBoard').value = '';
                document.getElementById('filterStatusBoard').value = '';
                document.getElementById('filterDateFromBoard').value = '';
                document.getElementById('filterDateToBoard').value = '';
                
                // Renderizar todas las tareas
                if (typeof window.renderKanbanTasks === 'function') {
                    window.renderKanbanTasks();
                }
                console.log('🧹 Filtros limpiados');
            });
            console.log('✅ Botón limpiar conectado');
        }
    }
    
    // Inicializar
    function init() {
        console.log('🚀 Iniciando parche...');
        
        // Esperar a que los proyectos estén disponibles
        const checkInterval = setInterval(function() {
            if (window.projects && window.currentProjectIndex !== undefined) {
                clearInterval(checkInterval);
                
                populateFilters();
                connectFilters();
                
                // Observar cambios de proyecto
                document.addEventListener('projectChanged', function() {
                    setTimeout(populateFilters, 200);
                });
                
                // Observar cuando se muestra el board
                const boardView = document.getElementById('boardView');
                if (boardView) {
                    const observer = new MutationObserver(function() {
                        if (boardView.style.display !== 'none') {
                            setTimeout(populateFilters, 300);
                        }
                    });
                    observer.observe(boardView, { attributes: true });
                }
                
                console.log('✅ PARCHE DE EMERGENCIA APLICADO');
            }
        }, 500);
    }
    
    // Ejecutar
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
    } else {
        init();
    }
    
    // Comando de diagnóstico
    window.diagnosticarFiltrosBoard = function() {
        console.group('🔍 DIAGNÓSTICO');
        const project = getProject();
        console.log('Proyecto:', project?.name);
        console.log('Tareas:', project?.tasks?.length);
        
        if (project?.tasks) {
            const asignados = [...new Set(project.tasks.map(t => t.assignee).filter(Boolean))];
            console.log('Asignados en datos:', asignados);
        }
        
        console.log('Select Asignado:', document.getElementById('filterAssigneeBoard')?.innerHTML);
        console.log('Valor actual:', document.getElementById('filterAssigneeBoard')?.value);
        console.groupEnd();
    };
    
    console.log('✅ Código de parche cargado');
})();

<script>
// ================================================
// 🎯 FILTROS DEL BOARD - VERSIÓN DEFINITIVA
// ================================================
(function() {
    console.log('🚀 Iniciando filtros del board...');
    
    // Función para cargar proyectos desde localStorage
    function cargarProyectos() {
        const projectsJSON = localStorage.getItem('projects');
        if (!projectsJSON) {
            console.log('⏳ Esperando projects en localStorage...');
            setTimeout(cargarProyectos, 1000);
            return;
        }
        
        try {
            // Cargar en window.projects
            window.projects = JSON.parse(projectsJSON);
            
            // Establecer índice actual
            const clienteId = localStorage.getItem('clienteId');
            if (clienteId) {
                const idx = window.projects.findIndex(p => p.clienteId === clienteId);
                window.currentProjectIndex = idx !== -1 ? idx : 0;
            } else {
                window.currentProjectIndex = 0;
            }
            
            console.log('✅ Proyectos cargados:', window.projects.length);
            console.log('📊 Proyecto actual:', window.projects[window.currentProjectIndex]?.name);
            
            // Configurar filtros
            configurarFiltros();
            
        } catch (e) {
            console.error('❌ Error al cargar proyectos:', e);
            setTimeout(cargarProyectos, 1000);
        }
    }
    
    // Función para configurar todos los filtros
    function configurarFiltros() {
        const project = window.projects[window.currentProjectIndex];
        if (!project || !project.tasks) {
            console.log('⏳ Proyecto sin tareas, reintentando...');
            setTimeout(configurarFiltros, 1000);
            return;
        }
        
        console.log('📋 Configurando filtros para:', project.name);
        console.log('📋 Tareas disponibles:', project.tasks.length);
        
        // ===== FILTRO POR ASIGNADO =====
        const selectAsignado = document.getElementById('filterAssigneeBoard');
        if (selectAsignado) {
            // Obtener asignados únicos
            const asignados = [...new Set(project.tasks
                .map(t => t.assignee)
                .filter(a => a && a.trim() !== '')
            )];
            
            console.log('👥 Asignados encontrados:', asignados);
            
            // Poblar select
            selectAsignado.innerHTML = '<option value="">Todos</option>';
            asignados.sort().forEach(asignado => {
                const option = document.createElement('option');
                option.value = asignado;
                option.textContent = asignado;
                selectAsignado.appendChild(option);
            });
            
            // Evento change
            selectAsignado.onchange = function() {
                const valor = this.value;
                console.log('🔍 Filtrando por asignado:', valor);
                
                // Guardar tareas originales
                const tareasOriginales = [...project.tasks];
                
                // Aplicar filtro
                if (valor) {
                    project.tasks = tareasOriginales.filter(t => t.assignee === valor);
                } else {
                    project.tasks = tareasOriginales;
                }
                
                // Renderizar
                if (typeof window.renderKanbanTasks === 'function') {
                    window.renderKanbanTasks();
                    console.log('✅ Board actualizado');
                }
                
                // Restaurar después de 5 segundos (para no perder datos)
                setTimeout(() => {
                    project.tasks = tareasOriginales;
                    if (typeof window.renderKanbanTasks === 'function') {
                        window.renderKanbanTasks();
                    }
                }, 5000);
            };
        }
        
        // ===== FILTRO POR PRIORIDAD =====
        const selectPrioridad = document.getElementById('filterPriorityBoard');
        if (selectPrioridad) {
            selectPrioridad.onchange = function() {
                const valor = this.value;
                console.log('🔍 Filtrando por prioridad:', valor);
                
                const tareasOriginales = [...project.tasks];
                
                if (valor) {
                    project.tasks = tareasOriginales.filter(t => t.priority === valor);
                } else {
                    project.tasks = tareasOriginales;
                }
                
                if (typeof window.renderKanbanTasks === 'function') {
                    window.renderKanbanTasks();
                }
                
                setTimeout(() => {
                    project.tasks = tareasOriginales;
                    if (typeof window.renderKanbanTasks === 'function') {
                        window.renderKanbanTasks();
                    }
                }, 5000);
            };
        }
        
        // ===== FILTRO POR ESTADO =====
        const selectEstado = document.getElementById('filterStatusBoard');
        if (selectEstado) {
            selectEstado.onchange = function() {
                const valor = this.value;
                console.log('🔍 Filtrando por estado:', valor);
                
                const tareasOriginales = [...project.tasks];
                
                if (valor) {
                    project.tasks = tareasOriginales.filter(t => t.status === valor);
                } else {
                    project.tasks = tareasOriginales;
                }
                
                if (typeof window.renderKanbanTasks === 'function') {
                    window.renderKanbanTasks();
                }
                
                setTimeout(() => {
                    project.tasks = tareasOriginales;
                    if (typeof window.renderKanbanTasks === 'function') {
                        window.renderKanbanTasks();
                    }
                }, 5000);
            };
        }
        
        // ===== BOTÓN LIMPIAR =====
        const btnLimpiar = document.getElementById('clearFiltersBoard');
        if (btnLimpiar) {
            btnLimpiar.onclick = function() {
                console.log('🧹 Limpiando todos los filtros');
                
                document.getElementById('filterAssigneeBoard').value = '';
                document.getElementById('filterPriorityBoard').value = '';
                document.getElementById('filterStatusBoard').value = '';
                document.getElementById('filterDateFromBoard').value = '';
                document.getElementById('filterDateToBoard').value = '';
                
                // Restaurar tareas originales
                const tareasOriginales = [...project.tasks];
                project.tasks = tareasOriginales;
                
                if (typeof window.renderKanbanTasks === 'function') {
                    window.renderKanbanTasks();
                }
            };
        }
        
        // ===== BOTÓN APLICAR =====
        const btnAplicar = document.getElementById('applyFiltersBoard');
        if (btnAplicar) {
            btnAplicar.onclick = function() {
                console.log('🎯 Aplicando todos los filtros');
                
                const filtroAsignado = document.getElementById('filterAssigneeBoard').value;
                const filtroPrioridad = document.getElementById('filterPriorityBoard').value;
                const filtroEstado = document.getElementById('filterStatusBoard').value;
                
                const tareasOriginales = [...project.tasks];
                
                // Aplicar todos los filtros
                let tareasFiltradas = tareasOriginales;
                
                if (filtroAsignado) {
                    tareasFiltradas = tareasFiltradas.filter(t => t.assignee === filtroAsignado);
                }
                
                if (filtroPrioridad) {
                    tareasFiltradas = tareasFiltradas.filter(t => t.priority === filtroPrioridad);
                }
                
                if (filtroEstado) {
                    tareasFiltradas = tareasFiltradas.filter(t => t.status === filtroEstado);
                }
                
                project.tasks = tareasFiltradas;
                
                if (typeof window.renderKanbanTasks === 'function') {
                    window.renderKanbanTasks();
                }
                
                setTimeout(() => {
                    project.tasks = tareasOriginales;
                    if (typeof window.renderKanbanTasks === 'function') {
                        window.renderKanbanTasks();
                    }
                }, 5000);
            };
        }
        
        console.log('✅ Todos los filtros configurados correctamente');
    }
    
    // Iniciar cuando el DOM esté listo
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', cargarProyectos);
    } else {
        cargarProyectos();
    }
    
    // También ejecutar cada vez que se muestre el board
    const boardView = document.getElementById('boardView');
    if (boardView) {
        const observer = new MutationObserver(function() {
            if (boardView.style.display !== 'none') {
                console.log('👁️ Board visible, recargando filtros...');
                setTimeout(cargarProyectos, 500);
            }
        });
        observer.observe(boardView, { attributes: true });
    }
    
    console.log('✅ Script de filtros instalado');
})();
</script>
